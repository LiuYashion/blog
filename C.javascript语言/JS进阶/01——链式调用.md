## 链式调用
编写方法实现下列功能
```js
// consle.log() 会默认调用参数的toString方法
console.log( add(1)(2)(3) );      //6
console.log( add(1)(2)(3)(4) );   //10

function add(x) {
  var sum = x;
  var tmp = function (y) {
    sum = sum + y;
    return tmp;
  };
  tmp.toString = function () {
    return sum;
  };
  return tmp;
}
```





```js
function ClassA(){
    this.prop1 = null;
    this.prop2 = null;
    this.prop3 = null;
}
ClassA.prototype = {
    method1 : function(p1){
        this.prop1 = p1;
        return this;
    },
    method2 : function(p2){
        this.prop2 = p2;
        return this;
    },
    method3 : function(p3){
        this.prop3 = p3;
        return this;
    }
}


var obj = new ClassA();  
obj.method1(1).method2(2).method3(3); // obj -> prop1=1,prop2=2,prop3=3  
```
