# 新建之后
先整合好reducer, 创建store放在全局下
```js
const store = createStore(reducer)

/** store的结构
  {
    dispatch: ƒ, 
    subscribe: ƒ, 
    getState: ƒ, 
    replaceReducer: ƒ, 
    Symbol(observable): ƒ
  }
 */

render(
  <Provider store={store}>
    <App/>
  </Provider>,
  document.getElementById('root')
)
```
